    ```typescript
    import Anthropic from '@anthropic-ai/sdk';
    import type { LLMProvider, CallOptions, LLMResponse } from './index.js';

    export class AnthropicProvider implements LLMProvider {
      public readonly name = 'anthropic';
      private client: Anthropic;
      private model: string;

      constructor(apiKey: string, model: string) {
        this.client = new Anthropic({ apiKey });
        this.model = model;
      }

      async call(systemPrompt: string, userPrompt: string, options?: CallOptions): Promise<LLMResponse> {
        const temperature = options?.temperature ?? 0.2;

        // Anthropic has no native JSON mode — use prompt engineering
        let finalSystemPrompt = systemPrompt;
        if (options?.responseFormat === 'json') {
          finalSystemPrompt += '\n\nIMPORTANT: Respond ONLY with valid JSON. No explanation, no markdown fences, no text before or after the JSON.';
        }

        const message = await this.client.messages.create({
          model: this.model,
          max_tokens: options?.maxTokens ?? 8096,
          temperature,
          system: finalSystemPrompt,
          messages: [
            { role: 'user', content: userPrompt },
          ],
        });

        // Extract text from content blocks
        const content = message.content
          .filter((block): block is Anthropic.TextBlock => block.type === 'text')
          .map((block) => block.text)
          .join('');

        return {
          content,
          usage: {
            inputTokens: message.usage.input_tokens,
            outputTokens: message.usage.output_tokens,
          },
          model: message.model,
        };
      }
    }
    ```

    KEY IMPLEMENTATION NOTES:
    - import type for interfaces to avoid runtime import
    - Anthropic SDK constructor takes { apiKey }
    - system prompt is a top-level parameter, not in messages array
    - messages array has { role: 'user', content: userPrompt }
    - JSON mode: append prompt engineering text to system prompt (no native response_format)
    - Content is ContentBlock[] — filter for type === 'text', map to .text, join
    - Usage: message.usage.input_tokens -> inputTokens, message.usage.output_tokens -> outputTokens
    - Model: message.model returns the actual model used
    - Default max_tokens: 8096 (reasonable default for analysis)
    - Default temperature: 0.2 per spec
    - Errors propagate naturally (no catch) — retry logic added in Phase 7

    ```typescript
    import OpenAI from 'openai';
    import type { LLMProvider, CallOptions, LLMResponse } from './index.js';

    export class OpenAIProvider implements LLMProvider {
      public readonly name = 'openai';
      private client: OpenAI;
      private model: string;

      constructor(apiKey: string, model: string) {
        this.client = new OpenAI({ apiKey });
        this.model = model;
      }

      async call(systemPrompt: string, userPrompt: string, options?: CallOptions): Promise<LLMResponse> {
        const temperature = options?.temperature ?? 0.2;

        // OpenAI has native JSON mode via response_format
        let finalSystemPrompt = systemPrompt;
        if (options?.responseFormat === 'json') {
          // JSON mode requires "json" to appear in messages
          finalSystemPrompt += '\n\nRespond with valid JSON only.';
        }

        const completion = await this.client.chat.completions.create({
          model: this.model,
          messages: [
            { role: 'system', content: finalSystemPrompt },
            { role: 'user', content: userPrompt },
          ],
          temperature,
          max_tokens: options?.maxTokens ?? 8096,
          ...(options?.responseFormat === 'json'
            ? { response_format: { type: 'json_object' as const } }
            : {}),
        });

        return {
          content: completion.choices[0]?.message.content ?? '',
          usage: {
            inputTokens: completion.usage?.prompt_tokens ?? 0,
            outputTokens: completion.usage?.completion_tokens ?? 0,
          },
          model: completion.model,
        };
      }
    }
    ```

    KEY IMPLEMENTATION NOTES:
    - OpenAI SDK constructor takes { apiKey }
    - System and user prompts are separate messages in the messages array
    - JSON mode: set response_format: { type: 'json_object' } AND append JSON reminder to system prompt
      (OpenAI requires "json" to appear in a message when using JSON mode)
    - Usage: completion.usage?.prompt_tokens -> inputTokens, completion.usage?.completion_tokens -> outputTokens
      (usage is optional in the type, so use nullish coalescing ?? 0)
    - Content: completion.choices[0]?.message.content ?? '' (safe access with nullish coalescing)
    - Model: completion.model returns the actual model string
    - Default max_tokens: 8096
    - Default temperature: 0.2 per spec

    2. Implement GoogleProvider class:

    ```typescript
    import { GoogleGenAI } from '@google/genai';
    import type { LLMProvider, CallOptions, LLMResponse } from './index.js';

    export class GoogleProvider implements LLMProvider {
      public readonly name = 'google';
      private ai: GoogleGenAI;
      private model: string;

      constructor(apiKey: string, model: string) {
        this.ai = new GoogleGenAI({ apiKey });
        this.model = model;
      }

      async call(systemPrompt: string, userPrompt: string, options?: CallOptions): Promise<LLMResponse> {
        const temperature = options?.temperature ?? 0.2;

        const response = await this.ai.models.generateContent({
          model: this.model,
          contents: userPrompt,
          config: {
            systemInstruction: systemPrompt,
            temperature,
            maxOutputTokens: options?.maxTokens ?? 8096,
            ...(options?.responseFormat === 'json'
              ? { responseMimeType: 'application/json' }
              : {}),
          },
        });

        return {
          content: response.text ?? '',
          usage: {
            inputTokens: response.usageMetadata?.promptTokenCount ?? 0,
            outputTokens: response.usageMetadata?.candidatesTokenCount ?? 0,
          },
          model: this.model,  // Google response does not include model name; use constructor value
        };
      }
    }
    ```

    KEY IMPLEMENTATION NOTES:
    - Use `@google/genai` (NOT the deprecated @google/generative-ai)
    - Import { GoogleGenAI } from '@google/genai'
    - Constructor takes { apiKey }
    - System prompt goes in config.systemInstruction (not a top-level parameter)
    - User prompt is passed as `contents` (string)
    - JSON mode: set config.responseMimeType = 'application/json' (native JSON mode)
    - Usage metadata field names differ from other providers:
      - response.usageMetadata?.promptTokenCount -> inputTokens
      - response.usageMetadata?.candidatesTokenCount -> outputTokens
    - IMPORTANT: Google response object does NOT include the model name; must use this.model from constructor
    - Default maxOutputTokens: 8096
    - Default temperature: 0.2 per spec
    - Errors propagate naturally — retry logic added in Phase 7
    - Config is flat — all config options go directly in config: { ... }

          ```typescript
          export function createProvider(providerName: string, model: string): LLMProvider {
            const envVarName = providerApiKeyEnvVars[providerName];
            if (!envVarName) {
              throw new ProviderDetectionError(
                `Unknown provider '${providerName}'. Supported providers: anthropic, openai, google.`
              );
            }
            const apiKey = process.env[envVarName];
            if (!apiKey) {
              throw new ProviderDetectionError(
                `Provider '${providerName}' selected but ${envVarName} is not set.`
              );
            }
            switch (providerName) {
              case 'anthropic':
                return new AnthropicProvider(apiKey, model);
              case 'openai':
                return new OpenAIProvider(apiKey, model);
              case 'google':
                return new GoogleProvider(apiKey, model);
              default:
                throw new ProviderDetectionError(
                  `Unknown provider '${providerName}'. Supported providers: anthropic, openai, google.`
                );
            }
          }
          ```
